CREATE TABLE IF NOT EXISTS ${constants.species_table} (
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
species VARCHAR(255), 
genus VARCHAR(255), 
family VARCHAR(255), 
order VARCHAR(255), 
class VARCHAR(255), 
phylum VARCHAR(255), 
kingdom VARCHAR(255), 
domain VARCHAR(255), 
status VARCHAR(100)
);

CREATE TABLE IF NOT EXISTS ${constants.animals_table} (
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
name VARCHAR(255), 
photo BLOB, 
location JSON, 
species_id INT, // FK
dob DATE, 
status VARCHAR(100),
gender VARCHAR(100),
description TEXT
);

CREATE TABLE IF NOT EXISTS ${constants.location_table} (
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
name VARCHAR(255)
);

CREATE TABLE IF NOT EXISTS ${constants.coordinates_table} (
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
location_id INT, // FK
coordinate JSON
);
